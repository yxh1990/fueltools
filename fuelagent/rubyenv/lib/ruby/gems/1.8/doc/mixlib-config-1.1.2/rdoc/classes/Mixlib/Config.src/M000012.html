    def method_missing(method_symbol, *args)
      num_args = args.length
      # Setting
      if num_args > 0
        method_symbol = $1.to_sym unless (method_symbol.to_s =~ /(.+)=$/).nil?
        internal_set method_symbol, (num_args == 1 ? args[0] : args)
      end
      
      # Returning
      self.configuration[method_symbol]        

    end
