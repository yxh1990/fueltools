    def config_attr_writer(method_symbol, &blk)
      meta = class << self; self; end
      method_name = "#{method_symbol.to_s}=".to_sym
      meta.send :define_method, method_name do |value|
        self.configuration[method_symbol] = blk.call(value)
      end
    end
